//Include jquery, console.js and console.css

//Assuming you have the following HTML
<button class="class1" id="clickMe">Click me!</button>
<div class="class2" style="color:white;background:red;width:200px;height:50px;">Mouse over!!!</div>

//Define which elements under which events should be logged
var config = {"events":{
			"click": ["#clickMe"],
			"mouseover": [".class2"]
		}
};

//Then initialize the Console once the body is loaded.
Console.init("#mylog", "body", config);

//You can also specify logging programatically
Console.log("first log");
Console.log("second, with ID", "ID01");
for(var i=0; i < 3; i++){
	Console.log("I will log "+(2-i)+" more events", "Iterator");
}
