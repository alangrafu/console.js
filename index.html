//Include jquery, console.js and console.css

//First config the elements you need to log
var config = 
{"eventsByClass":{
	"onclick": [{"class1": "message1"}, {"class2": "message2"}],
	"onmouseover": [{"class3": "message3"}]
}};

//Then initialize the Console once the body is loaded.
Console.init("#mylog", "body", config);
Console.log("first log");
Console.log("second, with ID", "ID01");
for(var i=0; i < 3; i++){
	Console.log("I will log "+(2-i)+" more events", "Iterator");
}
